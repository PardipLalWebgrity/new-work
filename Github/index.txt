git config --global core.editor "code --wait" = Default Text editor VS Code
git config --global core.editor = Check default text editor
git --version = Check if Git is in PATH
git config --local user.name "Your Name" = name will be attached to your commits
git config --global user.name "Your Name" = name will be attached to your commits
git config --system user.name "Your Name" = name will be attached to your commits
git config --global user.email "you@example.com" = email is also attached to your commits
git config --global user.name = Check global Git username
git config --global user.email = Check global email
git config --list = See all Git config settings
git config --global init.defaultBranch main = initial branch name all newly created repositories.
git init = create a new Git repository
git status = Check which files are tracked
git add <file> = To add a file to the staging area
git add --all or git add -A = Stage all changes
git status - See what is staged
git restore --staged <file> = bring back the last committed version of file
git reset <file> = remove file from staging
git commit -m "message" - Commit staged changes with a message
git commit -a -m "message" = stages all modified and deleted files.Then commits with message.
git log = See commit history
git log -10 = Last 10 commits
git log <file> = Show commits by one file
git log --author="Pardip" = Show commits by one author
git log --since="2024-01-01" --until="2024-06-01" = Show between dates
git commit --allow-empty -m "Start project" = Create an empty commit
git commit --amend --no-edit = update last commit, can't change the message
git commit --amend = it replaces the previous commit with same message
git commit --amend -m "new message" = Update last commit message
git reset --soft HEAD~1 = Undo last commit



ls = List files in the folder
pwd = Print Working Directory.
